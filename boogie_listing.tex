%% from Gemini

\usepackage{listings}
\usepackage{xcolor}


%% Define SMT-LIB language for listings 
\lstdefinelanguage{SMT}{
  % Main keywords
  keywords={declare-sort, declare-fun, assert},
  keywordstyle=\color{blue}\bfseries,
  % Verification keywords
  keywords=[2]{forall, let},
  keywordstyle=[2]{\color{violet}\bfseries},
  % Types
  keywords=[3]{bool, int, real, bv, T@U, T@T},
  keywordstyle=[3]{\color{green!40!blue}\ttfamily},
  % remaining
  sensitive=true,
  morecomment=[l]{;},
  morestring=[b]",
  literate=%
    % SMT functions
    {=}{{{\color{yellow!40!black}{=}}}}1
    {=>}{{{\color{yellow!40!black}{=>}}}}2
}


%% Define Boogie language for listings 
\lstdefinelanguage{Boogie}{
  % Main keywords
  keywords={procedure, implementation, function, method,
  type, const, axiom, forall, var,
  lambda, true, false,
  returns, invariant, ensures, requires, modifies, free, where},
  keywordstyle=\color{blue}\bfseries,
  % Verification keywords
  keywords=[2]{assert, assume, havoc, if, then, else, while, call, return},
  keywordstyle=[2]{\color{violet}\bfseries},
  % Types
  keywords=[3]{bool, int, real, bv},
  keywordstyle=[3]{\color{green!40!blue}\ttfamily},
  % remaining
  sensitive=true,
  morecomment=[l]{//},
  morecomment=[s]{/*}{*/},
  morestring=[b]"
}
