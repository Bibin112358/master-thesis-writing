%% from Gemini

\usepackage{listings}
\usepackage{xcolor}

%% Define Boogie language for listings 
\lstdefinelanguage{Boogie}{
  % Main keywords
  keywords={procedure, implementation, function,
  type, const, axiom, forall, var,
  returns, invariant, ensures, requires, modifies, free},
  keywordstyle=\color{blue}\bfseries,
  % Verification keywords
  keywords=[2]{assert, assume, havoc, if, else, while, call, return},
  keywordstyle=[2]{\color{violet}\bfseries},
  % Types
  keywords=[3]{bool, int, real, bv},
  keywordstyle=[3]{\color{green!40!blue}\ttfamily},
  % remaining
  sensitive=true,
  morecomment=[l]{//},
  morecomment=[s]{/*}{*/},
  morestring=[b]",
}
