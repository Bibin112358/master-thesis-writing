\subsection{Type of Val for Maps}

\begin{verbatim}
(* Semantics.thy *)
text<Type interpretation:
Each value of the abstract carrier type must be mapped to a corresponding type (constructed via a
type constructor).>

class absval =
  fixes absval_ty :: "'a => (tcon_id x ty list)"

class mapval =
  fixes mapval_ty :: "'a => (ty x ty)"

fun type_of_val :: "('a::absval, 'm::mapval) val => ty"
  where
   "type_of_val (LitV v) = TPrim (type_of_lit v)"
 | "type_of_val (AbsV v) = TCon (fst (absval_ty v)) (snd (absval_ty v))"
 | "type_of_val (MapV v) = TMap (fst (mapval_ty v)) (snd (mapval_ty v))"

fun tmap_lvl :: "ty => nat" where
    "tmap_lvl (TMap tk tv) = max (1 + tmap_lvl tk) (tmap_lvl tv)"
  | "tmap_lvl (TCon _ ty_args) = 0"
  | "tmap_lvl (TVar i) = 0"
  | "tmap_lvl _ = 0"


(* Map3Semantics.thy *)
subsection <Type Of Val>

primrec tyL where "tyL (FunL f tk tv) = (tk, tv)"

fun ty321 :: "'a val321 => ty x ty" where
    "ty321 (Inr (Inr m)) = tyL m"
  | "ty321 (Inr (Inl m)) = tyL m"
  | "ty321 (Inl m) = tyL m"

instantiation L :: (type) mapval begin
  fun mapval_ty_L where "mapval_ty_L x = tyL x"
  instance .. end

instantiation sum :: (mapval, mapval) mapval begin
  primrec mapval_ty_sum :: "'a + 'b => ty x ty" where
      "mapval_ty_sum (Inl x) = mapval_ty x"
    | "mapval_ty_sum (Inr x) = mapval_ty x"
  instance .. end

lemma mapval_ty_eq_ty321: "mapval_ty = ty321"
  apply (rule, rename_tac x)
  by (case_tac x rule: ty321.cases; simp)

fun key_ty where "key_ty (TMap tk _) = tk" | "key_ty _ = undefined"
fun val_ty where "val_ty (TMap _ tv) = tv" | "val_ty _ = undefined"

primrec wf_L where
  "wf_L n (FunL _ tk tv) = (
    (tmap_lvl tk <= n-1) AND (tmap_lvl tv <= n) AND
    ((tmap_lvl tk = n-1) OR (tmap_lvl tv = n)))"  (* guarantee highest key or value on this level *)

fun wf_ty :: "'a valn => bool" where
    "wf_ty (LitV v) = True"
  | "wf_ty (AbsV v) = True"
  | "wf_ty (MapV (Inr (Inr m))) = wf_L 1 m"
  | "wf_ty (MapV (Inr (Inl m))) = wf_L 2 m"
  | "wf_ty (MapV (Inl m)) = wf_L 3 m"

\end{verbatim}
